Endpoints disponibles :
```javascript
// Démarrage écosystème
POST /api/ecosystem?action=start

// Prospection intelligente
POST /api/ecosystem?action=prospection
{
  "industry": "technology",
  "company_size": "50-500",
  "location": "France"
}

// Campagne multi-canal
POST /api/ecosystem?action=campaign
{
  "name": "AI Campaign",
  "channels": ["email", "linkedin"],
  "strategy": ["nurturing"]
}

// Expérimentation créative
POST /api/ecosystem?action=experiment
{
  "experiment_type": "ab_test",
  "creativity_level": "high"
}

// Status et métriques
GET /api/ecosystem?action=status
GET /api/ecosystem?action=metrics
GET /api/ecosystem?action=health
```

### Utilisation Programmatique
```javascript
const { EcosystemLauncher } = require('./src/EcosystemLauncher');

// Démarrage express
const ecosystem = await EcosystemLauncher.launch();

// Prospection intelligente
const prospects = await ecosystem.runIntelligentProspection({
    industry: 'technology',
    company_size: '50-500'
});

// Campagne orchestrée
const campaign = await ecosystem.runSmartCampaign({
    channels: ['email', 'linkedin'],
    strategy: ['aggressive', 'nurturing']
});
```
